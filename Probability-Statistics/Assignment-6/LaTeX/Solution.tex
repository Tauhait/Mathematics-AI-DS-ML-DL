\subsection*{\boldsymbol{Problem\ 6.8}}
\begin{flushleft} A and B are two events such that P (A) $\neq$  0.
\newline Find P ( B / A), if
\newline (i) A $\subset$ B
\newline (ii) A $\cap$ B = $\phi$ \end{flushleft}

\subsection*{\boldsymbol{Solution}}\\
\begin{flushleft}
Given P (A) $\neq$  0.\newline We know,
\end{flushleft}
\begin{align}
P\ (\ B/A\ )\ =\frac{P\ (A\ \cap\ B)}{P\ (A)}
\end{align}
(i) If A $\subset$ B
\begin{figure}[h!]
    \includegraphics[width=7.5cm]{Assignment-6/Figures/a_subset_b.JPG}
    \centering \caption*{Fig 1.1: A \subset B}
\end{figure}
\newline
\implies A\ $\cap$ B = A
\newline
\implies P\ (A\ $\cap$ B) = P (A)
\newline
\newline
Putting P (A $\cap$ B) in (0.0.1), we get
\newline
\begin{align*}
P\ (\ B/A\ )\ =\ \frac{P\ (A)}{P\ (A)} = 1
\end{align*}
\newline

(ii) If A $\cap$ B = $\phi$
\begin{figure}[h!]
    \includegraphics[width=7.5cm]{Assignment-6/Figures/a_interset_b_zero.JPG}
    \centering \caption*{Fig 1.2: A $\cap$ B = $\phi$}
\end{figure}
\newline
\implies P\ (A\ $\cap$ B) = 0
\newline
\newline
Putting P (A\ $\cap$ B) in (0.0.1), we get
\newline
\begin{align*}
P\ (\ B/A\ )\ =\ \frac{0}{P\ (A)} = 0
\end{align*}
